\question{}
יהי שדה $\FF$ ומטריצות $A, B \in M_n(\FF)$ חופפות.

\subquestion{}
נראה כי אם $A$ סימטרית אז $B$ סימטרית.
\begin{proof}
	נניח ש־$P \in M_n(\FF)$ מטריצה הפיכה המקיימת $A = P^t B P$.
	נסמן $g : \FF^n \times \FF^n \to \FF$ התבנית בילינארית הסימטרית המקיימת ${[g]}_E = A$, אז הוכחנו בכיתה כי ${[g]}_P = B$.
	בבניית $g$ השתמשנו בהנחה ש־$A$ סימטרית, ומהנתון החדש על $B$ נקבל שגם היא סימטרית.
\end{proof}

\subquestion{}
נראה כי אם $A$ הפיכה אז $B$ הפיכה.
\begin{proof}
	נניח ש־$P \in M_n(\FF)$ מטריצה הפיכה המקיימת $B = P^t A P$.
	$P^t$ הפיכה אף היא משיקולי דרגה, וכפל של הפיכות משמר הפיכות, לכן גם $A P$ וכן $ A^t (A P)$ הפיכות ובפרט $B$ הפיכה.
\end{proof}

\subquestion{}
נוכיח שקיים $0 \ne c \in \FF$ כך ש־$\det B = c^2 \det A$.
\begin{proof}
	%מהסעיף הקודם נסיק ש־$A$ הפיכה אם ורק אם $B$ הפיכה, ולכן אם $A$ לא הפיכה ו־$\det A = 0$ אז גם $\det B = 0$ ונוכל לבחור $c = 1$.
	נסמן שוב $A = P^t B P$, אז,
	\[
		\det A
		= \det(P^t B P)
		= (\det P^t) (\det B) (\det P)
		= c \cdot \det B \cdot c
		= c^2 \det B
	\]
	כאשר $c = \det P$.
\end{proof}

\question{}
נגדיר את המטריצות,
\[
	A = \begin{pmatrix}
		1 & 0 \\
		0 & 2
	\end{pmatrix},
	I_2 \in M_2(\FF)
\]
ונבדוק אם הן חופפות עבור הגדרות $\FF$.

\subquestion{}
נגדיר $\FF = \RR$ ונראה שהן חופפות.
\begin{proof}
	נבחין כי $2 = 1 \cdot \sqrt{2}^2$ ולכן $A$ חופפת ל־$I_2$ עם הבסיס $(1, 0), (0, \frac{1}{\sqrt{2}})$ מטענה על חפיפה לתת־קבוצת כיסוי.
\end{proof}

\subquestion{}
נגדיר $\FF = \QQ$ ונראה שהן לא חופפות.
\begin{proof}
	אילו הן היו חופפות אז היה באופן דומה לסעיף הקודם מתקיים $2 = 1 \cdot q^2$ עבור $q \in \QQ$ כלשהו, כלומר נקבל $\sqrt{2} \in \QQ$ בסתירה.
\end{proof}

\subquestion{}
נגדיר $\FF = \FF_3$ ונראה שהן לא חופפות.
\begin{proof}
	המטריצות חופפות אם קיים $c \in \FF_3^*$ המקיים $2 = 1 \cdot c^2$, אבל $c = 1 \implies c^2 = 1$ וכן $c = 2 \implies c^2 = 1$ ולכן אין שורש ל־$2$ ובהתאם אין חפיפה.
	נבחין כי $|\FF_3^*| = 2$ ולכן $R = \{1, 2\}$ היא קבוצת הנציגים היחידה שקיימת ונקבל שוב שאין חפיפה.
\end{proof}

\subquestion{}
עבור $\FF = \FF_7$ נראה חפיפה.
\begin{proof}
	נבחין כי $c = 3$ הוא איבר שונה מאפס המקיים $c^2 =_{\FF_7} 2$ ולכן $2 = 1 \cdot c^2$ ונקבל ש־$A$ חופפת ל־$I_2$ עם הבסיס $(1, 0), (0, \frac{1}{3})$.
\end{proof}

\question{}
יהי $V$ מרחב וקטורי מעל $\FF$ ו־$g : V \times V \to \FF, q : V \to \FF$ פונקציות כך שמתקיים,
\[
	\forall v, w \in V,\ 
	g(v, w)
	= \frac{1}{2} (q(v + w) - q(v) - q(w))
\]
נראה שאם $g$ תבנית בילינארית סימטרית וכן $q(c v) = c^2 q(v)$ לכל $c \in \FF, v \in V$ אז $q$ תבנית ריבועית.
\begin{proof}
	נראה טענה מעט חזקה יותר, נראה ש־$q$ היא התבנית הריבועית הנוצרת על־ידי $g$, כלומר,
	\[
		\forall v \in V,\ 
		q(v) = g(v, v)
	\]
	נציב,
	\[
		g(v, v)
		= \frac{1}{2}( q(v + v) - q(v) - q(v))
		\overset{(1)}{=}  \frac{1}{2} (4q(v) - q(v) - q(v))
		= q(v)
	\]
	כאשר $(1)$ נובע מהנתון אודות ההומוגניות של $q$.
	קיבלנו שלפי הגדרה $q$ אכן התבנית הריבועית המתאימה ל־$g$.
\end{proof}

\question{}
נמצא דוגמה למרחב וקטורי $V$ מעל $\FF$ ופונקציות $g : V \times V \to \FF, q : V \to \FF$ כך שמתקיים,
\[
	\forall v, w \in V,\ 
	g(v, w)
	= \frac{1}{2} (q(v + w) - q(v) - q(w))
\]
ו־$g$ תבנית בילינארית סימטרית אבל ש־$q$ איננה תבנית ריבועית.
\begin{solution}
	נגדיר $\FF = \FF^3$ וכן $V = \FF^1$, $q(x) = x^3$.
	בהתאם מתקיים,
	\[
		q(v + w)
		= {(v + w)}^3
		= v^3 + 3 (\cdots) + w^3
		= v^3 + w^3
		= q(v) + q(w)
	\]
	ובהתאם $q(v + w) - q(v) - q(w) = 0$, נקבל אם כך ש־$g(v, w) = 0$ התבנית הקבועה מקיימת את הטענה עבור $q$ שהגדרנו.
	בהתאם אם $q$ הייתה תבנית ריבועית אז היינו מקבלים $1 = 1^3 = q(1) = g(1, 1) = 0$ בסתירה.
\end{solution}

\question{}
יהי $V$ מרחב וקטורי מעל $\FF$ ו־$q : V \to \FF$ תבנית ריבועית, ויהיו $v, w \in V$ כך ש־$q(v) \ne 0$. \\
נראה שקיים $c \in \FF$ יחיד כך ש־$w - c v \in {\{ v \}}^{\perp}$ ונמצא נוסחה מפורשת ל־$c$ כתלות ב־$q, v, w$.
\begin{proof}
	נגדיר $g : V \times V \to \FF$ התבנית הבילינארית היחידה המתאימה ל־$q$, ולכן $g(v, v) \ne 0$.
	מתקיים,
	\[
		w - cv \in {\{ v \}}^\perp
		\iff g(w - cv, v) = 0
		\iff g(w, v) - c g(v, v) = 0
		\iff g(w, v) = c q(v)
	\]
	נגדיר אם כך $c = \frac{g(w, v)}{q(v)}$, הגדרה זו תקפה שכן $q(v) \ne 0$.
	משרשרת הגרירות נקבל שגם $w - c v \in {\{ v \}}^\perp$ כפי שרצינו, ומנוסחה מפורשת זו נקבל גם יחידות.
\end{proof}

\question{}
נמצא דוגמה למרחב וקטורי $V$ מעל $\FF$, תבנית ריבועית $q : V \to \FF$ ווקטורים $v, w \in V$ כך ש־$w - c v \notin {\{ v \}}^\perp$ לכל $c \in \FF$.
\begin{solution}
	נגדיר $\FF = \RR, V = \RR^2$ וכן $q(\begin{pmatrix} x_1 \\ x_2 \end{pmatrix}) = x_1^2 - x_2^2$, ראינו בשיעור כי זוהי אכן תבנית ריבועית. \\
	נקבל שגם $g(\begin{pmatrix} x_1 \\ x_2 \end{pmatrix}, \begin{pmatrix} y_1 \\ y_2 \end{pmatrix}) = x_1 y_1 - x_2 y_2$ היא התבנית הבילינארית הסימטרית המתאימה לה.
	נגדיר $v = (1, -1)$ ולכן $q(v) = 0$, ונגדיר את $w = (1, 0)$, נקבל,
	\[
		w - c v
		= (1, 0) - c (1, -1)
		= (1 - c, c)
	\]
	ובהתאם,
	\[
		g(v, w - c v)
		= g((1, -1), (1 - c, c))
		= 1 \cdot (1 - c) - (-1) \cdot c
		= 1 - c + c
		= 1
	\]
	כלומר מצאנו שתמיד $g(v, w - c w) \ne 0$, לכן $w - c v \notin {\{ v \}}^\perp$ לכל $c \in \FF$.
\end{solution}

\question{}
תהי $q : \RR^3 \to \RR$ המוגדרת על־ידי,
\[
	q(x_1, x_2, x_3)
	= x_1^2 - 4 x_1 x_2 + 3 x_2^2 - 6 x_2 x_3 - 9 x_3^2
.\]
נמצא בסיס $\Bb$ של $\RR^3$ עבורו ${[g_q]}_\Bb$ אלכסונית כך שאיבריה שייכים ל־$\{1, -1, 0\}$.
\begin{solution}
	נחפוף אלמנטרית, בכל מהלך נכתוב את פעולת השורה שנעשה אך נבצע גם את מקבילתה על העמודה וגם את מטריצת היחידה מתחת לה.
	המטריצה הראשונה היא המטריצה היחידה המתאימה בהתאם להסבר שנאמר בכיתה.
	\[
		\begin{pmatrix}
			1 & -2 & 0 \\
			-2 & 3 & -3 \\
			0 & -3 & -9 \\
			\hline
			1 & 0 & 0 \\
			0 & 1 & 0 \\
			0 & 0 & 1
		\end{pmatrix}
		\xrightarrow{R_2 \to R_2 + 2 R_1}
		\begin{pmatrix}
			1 & 0 & 0 \\
			0 & -1 & -3 \\
			0 & -3 & -9 \\
			\hline
			1 & 2 & 0 \\
			0 & 1 & 0 \\
			0 & 0 & 1
		\end{pmatrix}
		\xrightarrow{R_3 \to R_3 - 3 R_2}
		\begin{pmatrix}
			1 & 0 & 0 \\
			0 & -1 & 0 \\
			0 & 0 & 0 \\
			\hline
			1 & 2 & -6 \\
			0 & 1 & -3 \\
			0 & 0 & 1
		\end{pmatrix}
	.\]
	לכן בבסיס $\Bb = ({(1, 0, 0)}^t, {(2, 1, 0)}^t, {(-6, -3, 1)}^t)$ מתקיים ${[g_q]}_\Bb = \diag(1, -1, 0)$.
\end{solution}

\question{}
יהי $p > 3$ ראשוני כלשהו.
נראה שבשדה $\FF_p$ מתקיים $\prod_{n = 2}^{p - 2} n = 1$.
\begin{proof}
	נבחין שכל $1 < n < p - 1$ הוא הפיך, אם $n \ne m$ אז בהכרח גם $n^{-1} \ne m^{-1}$.
	נגדיר $A_1 = \{2, 2^{-1}\}$, אז $\prod_{n \in A_1} n = 1$.
	לכל $n \ge 1$ נגדיר $k = \min(\{2, 3, \ldots, p - 2\} \setminus A_n)$ וכן את $A_{n + 1} = A_n \cup \{ k, k^{-1} \}$, ובאינדוקציה נקבל ש־$|A_{n + 1}| = |A_n| + 2 = 2n$.
	בנוסף מהבנייה נקבל $\prod_{k \in A_n} k = 1$ ולכן $|A_{\frac{1}{2}(p - 2)}| = p - 2$ ובהתאם קיבלנו את הטענה הרצויה.
\end{proof}

\question{}
יהי $p > 3$ ראשוני ונוכיח שבשדה $\FF_p$ מתקיים,
\[
	\prod_{n = 1}^{p - 2} n
	= {(-1)}^{\frac{p - 3}{2}} {\left( \prod_{n = 1}^{\frac{p - 1}{2}} n \right)}^2
.\]
\begin{proof}
	לכל $1 \le n < p$ מתקיים,
	\[
		n (p - n)
		= n p - n^2
		= -n^2
		\tag{1}
	.\]
	וכן על־ידי שינוי סדר המכפלה,
	\[
		\prod_{n = 1}^{p - 2} n
		= \prod_{n = 2}^{\frac{p - 3}{2}} n (p - n)
		\overset{(1)}{=} \prod_{n = 2}^{\frac{p - 3}{2}} -n^2
		\overset{(2)}{=} {(-1)}^{\frac{p - 3}{2}} {\left( \prod_{n = 2}^{\frac{p - 3}{2}} n \right)}^2
	.\]
	כאשר $(2)$ נובע מהוצאת כפולות $-1$ וכללי חזקות למכפלות סופיות.
\end{proof}

\question{}
יהי $p$ ראשוני המקיים $p \equiv 1 \mod 4$.
נראה ש־$-1 \in {(\FF_p^*)}^2$.
\begin{proof}
	ניזכר כי $-1 \in {(\FF_p^*)}^2$ אם ורק אם קיים $0 \ne x \in \FF_p$ המקיים $x^2 = -1$.
	משתי השאלות הקודמות נקבל,
	\[
		1
		= \prod_{n = 1}^{p - 2} n
		= {(-1)}^{\frac{p - 3}{2}} {\left( \prod_{n = 2}^{\frac{p - 1}{2}} n \right)}^2
	.\]
	אבל $p - 3 \equiv 2 \mod 4$ על־פי הנתון והוספה לשני האגפים ולכן ${(-1)}^{\frac{p - 3}{2}} = -1$ בלבד.
	נסמן $c = {(2 \cdot 3 \cdots \frac{p - 1}{2})}^2$ ונקבל $1 = -c^2$, לאחר כפל $-1 = c^2$, כלומר $c$ מעיד על $-1 \in {(\FF_p^*)}^2$.
\end{proof}

\question{}
יהי $p > 2$ ראשוני, ונראה שאם $-1 \in {(\FF_p^*)}^2$ אז בשדה $\FF_p$ מתקיים,
\[
	{(-1)}^{\frac{p - 1}{2}} = 1
.\]
\begin{proof}
	נסמן $c \in \FF_p^*$ האיבר שמקיים $c^2 = -1$.
	ממשפט קושי לחבורות מתקיים $c^{|\FF_p^*|} = 1$, אבל $|\FF_p^*| = p - 1$ ולכן $c^{p - 1} = 1$.
	נבחין כי $p - 1$ וזגי ולכן מתקיים $p - 1 = 2 \cdot \frac{p - 1}{2}$, ובהתאם גם,
	\[
		1
		= c^{p - 1}
		= c^{2 \cdot \frac{p - 1}{2}}
		= {(-1)}^{\frac{p - 1}{2}}
	.\]
	וקיבלנו בדיוק את המבוקש.
\end{proof}

\question{}
יהי $p$ ראשוני המקיים $p \equiv 3 \mod 4$, נראה ש־$-1 \notin {(\FF_p^*)}^2$.
\begin{proof}
	נניח בשלילה ש־$1 \in {(\FF_p^*)}^2$ ולכן מהשאלה הקודמת נקבל ש־${(-1)}^{\frac{p - 1}{2}} = 1$.
	נבחין כי קיים $k \in \NN$ כך ש־$p = 3 + 4k$ ולכן גם $\frac{p - 1}{2} = \frac{2 + 4k}{2} = 1 + 2k$, כלומר $\frac{p - 1}{2} \equiv 1 \mod 2$.
	מאי־זוגיות נקבל ש־$1 = {(-1)}^{\frac{p - 1}{2}} = -1$, אבל מכך נובע ש־$p = 2$ בלבד, וכן $2 \equiv 2 \mod 4$ בסתירה.
\end{proof}
