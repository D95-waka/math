נוכיח באינדוקציה שלכל $n \in \NN$ מתקיים,
\[
	x^n - 1
	= (x - 1)(x^{n - 1} + \cdots + 1)
.\]
\begin{proof}
	נוכיח את הטענה באינדוקציה. \\
	עבור $n = 1$ נקבל $x - 1 = (x - 1)(1)$ כפי שרצינו. \\
	נניח עתה שעבור $n \in \NN$ כלשהו מתקיימת הטענה ונבדוק את $n + 1$.
	\[
		x^{n + 1} - 1
		= (x^{n + 1} - x) + (x - 1)
		= x (x - 1) (x^{n - 1} + \cdots + 1) + (x - 1)
		= (x - 1) (x^n + \cdots + x) + (x - 1)
		= (x - 1) (x^n + \cdots + x + 1)
	\]
	והטענה חלה.
\end{proof}

נראה שלכל $n \in \NN$ מתקיים,
\[
	\frac{1}{2} \cdot \frac{3}{4} \cdots \frac{2n - 1}{2n} < \frac{1}{\sqrt{2n + 1}}
\]
\begin{proof}
	עבור בסיס האינדוקציה נבדוק את $n = 1$,
	\[
		\frac{1}{2} < \frac{1}{\sqrt{3}}
		\iff \sqrt{3} < 2
		\iff 3 < 4
	.\]
	ולכן הטענה מתקיימת.

	נניח שהטענה נכונה עבור $n$ ונבדוק את המקרה $n + 1$,
	\[
		\frac{1}{2} \cdot \frac{3}{4} \cdots \frac{2n - 1}{2n} \frac{2n + 1}{2n + 2}
		< \frac{1}{\sqrt{2n + 1}} \frac{2n + 1}{2n + 2}
		= \frac{\sqrt{2n + 1}}{2n + 2}
		< \frac{\sqrt{2n + 2}}{2n + 2}
		= \frac{1}{\sqrt{2n + 2}}
	\]
	השלמנו את מהלך האינדוקציה ולכן הטענה נכונה לכל $n \in \NN$.
\end{proof}

נוכיח שאם $a_1 \cdots a_n = 1$ אז $(1 + a_1) \cdots (1 + a_n) \ge 2^n$.
\begin{proof}
	באינדוקציה.
	עבור $n = 1$ נקבל $a_1 = 1$ ולכן $(1 + a_1) \ge 1 + 1$.

	נניח שהטענה נכונה עבור $n$ ותהי סדרה $a_1 \cdots a_{n + 1} = 1$, אז בפרט גם $a_1 \cdots a_{n - 1} \cdot (a_n a_{n + 1}) = 1$ סדרה מאורך $n$ ולכן מתקיים,
	\[
		\overbrace{(1 + a_1) \cdots (1 + a_{n - 1})}^{\overset{\text{def}}{=} K} (1 + a_n a_{n + 1}) \ge 2^n \tag{1}
	\]
	נניח בלי הגבלת הכלליות ש־$(a_n)$ סדרה מונוטונית עולה, אחרת נוכל להגדיר מחדש סדרה $b_1 = \min\{ a_k \}$ ו־$b_k = \min \{ a_k \} \setminus \{ b_1, \ldots, b_{k - 1} \}$.
	אז בפרט מתקיים $a_n a_{n + 1} \ge 1$, וכן מתקיים,
	\[
		(1 + a_n) (1 + a_{n + 1})
		= 1 + a_n + a_{n + 1} + a_n a_{n + 1}
		\ge 4 \tag{2}
	\]
	נוכל להציג את $(1)$ גם על־ידי,
	\[
		K ((1 + a_n) (1 + a_{n + 1}) - a_n - a_{n + 1}) \ge 2^n
	\]
	ועל־ידי פתיחת סוגריים והעברת אגפים,
	\[
		\overbrace{K (1 + a_n) (1 + a_{n + 1})}^{\overset{\text{def}}{=} K'}
		\ge 2^n + (a_n + a_{n + 1}) K
		\overset{\text{הנחת האינדוקציה}}{\ge} 2^n + (a_n + a_{n + 1}) 2^{n - 1}
		\overset{(2)}{\ge} 2^n + 2 \cdot 2^{n - 1}
		= 2^{n + 1}
	\]
	ולכן הטענה נובעת.
\end{proof}
