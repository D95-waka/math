נוכיח באינדוקציה שלכל $n \in \NN$ מתקיים,
\[
	x^n - 1
	= (x - 1)(x^{n - 1} + \cdots + 1)
.\]
\begin{proof}
	נוכיח את הטענה באינדוקציה. \\
	עבור $n = 1$ נקבל $x - 1 = (x - 1)(1)$ כפי שרצינו. \\
	נניח עתה שעבור $n \in \NN$ כלשהו מתקיימת הטענה ונבדוק את $n + 1$.
	\[
		x^{n + 1} - 1
		= (x^{n + 1} - x) + (x - 1)
		= x (x - 1) (x^{n - 1} + \cdots + 1) + (x - 1)
		= (x - 1) (x^n + \cdots + x) + (x - 1)
		= (x - 1) (x^n + \cdots + x + 1)
	\]
	והטענה חלה.
\end{proof}

נראה שלכל $n \in \NN$ מתקיים,
\[
	\frac{1}{2} \cdot \frac{3}{4} \cdots \frac{2n - 1}{2n} < \frac{1}{\sqrt{2n + 1}}
\]
\begin{proof}
	עבור בסיס האינדוקציה נבדוק את $n = 1$,
	\[
		\frac{1}{2} < \frac{1}{\sqrt{3}}
		\iff \sqrt{3} < 2
		\iff 3 < 4
	.\]
	ולכן הטענה מתקיימת.

	נניח שהטענה נכונה עבור $n$ ונבדוק את המקרה $n + 1$,
	\[
		\frac{1}{2} \cdot \frac{3}{4} \cdots \frac{2n - 1}{2n} \frac{2n + 1}{2n + 2}
		< \frac{1}{\sqrt{2n + 1}} \frac{2n + 1}{2n + 2}
		= \frac{\sqrt{2n + 1}}{2n + 2}
		< \frac{\sqrt{2n + 2}}{2n + 2}
		= \frac{1}{\sqrt{2n + 2}}
	\]
	השלמנו את מהלך האינדוקציה ולכן הטענה נכונה לכל $n \in \NN$.
\end{proof}
