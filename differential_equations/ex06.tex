\subquestion{}
נגדיר,
\[
	A
	= \begin{pmatrix}
		-5 & 1 \\
		4 & -2
	\end{pmatrix}
\]
נמצא פתרון כללי למשוואה $x'(t) = A x(t)$ ונצייר דיאגרמות זרימה מתאימה.
נקבע אם הראשית היא נקודת שיווי משקל יציבה או לא.
\begin{solution}
	נגדיר $P_A(t) = \det(I t - A) = (t + 5)(t + 2) - 4 = (t + 6)(t + 1)$ פולינום אופייני ולכן $A$ לכסינה וכן $A = P^{-1} \diag(-6, -1) P$ לאיזושהי $P \in M_2(\RR)$ הפיכה.
	נחשב את $P$ על־ידי מציאת וקטורים עצמיים,
	\[
		(A + 6 I) u = 0
		\iff \begin{pmatrix}
			1 & 1 \\
			4 & 4
		\end{pmatrix} u = 0
		\iff u = \Sp\{ {(1, -1)}^t \}
	\]
	וכן,
	\[
		(A + I) u = 0
		\iff \begin{pmatrix}
			-4 & 1 \\
			4 & -1
		\end{pmatrix} u = 0
		\iff u = \Sp\{ {(1, 4)}^t \}
	\]
	כלומר אם נגדיר,
	\[
		P
		= \begin{pmatrix}
			1 & 1 \\
			-1 & 4
		\end{pmatrix}
	\]
	היא מטריצה הפיכה המלכסנת את $A$.
	ממשפט פתרון משוואות לינאריות נסיק שהפונקציה,
	\[
		x(t)
		= P^{-1} e^{\diag(-6, -1) t} P x_0
		= P^{-1} \diag(e^{-6 t}, e^{-t}) P x_0
	\]
	הוא הפתרון היחיד למשוואה $x'(t) = A x(t), x(0) = x_0$.

	נשרטט את הזרימה:
	\begin{otherlanguage}{english}
		\begin{center}
			\begin{tikzpicture}
				\begin{axis}[ymin=-4,
						ymax=4,
						xmin=-4,
						xmax=4,
						samples=50,
						axis y line=middle,
						axis x line=middle,
						restrict x to domain=-4:4,
						restrict y to domain=-4:4,
						clip=true]
					\foreach \k in {0,1,...,23} {
						\edef\X{cos(360 * \k / 24)}
						\edef\Y{2 * sin(360 * \k / 24)}
						\addplot [blue,domain=-2:2,samples=50,variable=\t] ({exp(-6 * \t) * \X}, {exp(-1 * \t) * \Y});
					}
				\end{axis}
			\end{tikzpicture}
		\end{center}
	\end{otherlanguage}
	כאשר $x$ מתרחק מ־$0$ בזמן חיובי ומתקרב ל־$0$ בזמן שלילי. \\
	בהתאם נסיק ש־$0$ אכן נקודת שיווי משקל לא יציבה בזמן חיובי ויציבה בזמן שלילי.
\end{solution}

\subquestion{}
נמיר את המשוואה $x'' + x' - 2x = 0$ במערכת משוואות מסדר ראשון, נמצא פתרון ונצייר דיאגרמה.
\begin{solution}
	אנו יודעים כי ניתן להמיר על־ידי המטריצה,
	\[
		A = \begin{pmatrix}
			0 & 1 \\
			2 & -1
		\end{pmatrix}
	\]
	הפעם נקבל,
	\[
		P_A(t)
		= t (t + 1) - 2
		= t^2 + t - 2
		= (t - 1)(t + 2)
	\]
	לנוחות הקורא נדלג על חישוב הווקטורים העצמיים ונכריז שעבור,
	\[
		P = \begin{pmatrix}
			1 & 1 \\
			1 & -2
		\end{pmatrix}
	\]
	מקיימת $A = P^{-1} \diag(1, -2) P$.
	אז נקבל שהפתרון של $y'(t) = A y(t), y(0) = y_0$ הוא,
	\[
		y(t)
		= P^{-1} \diag(e^t, e^{-2t}) P y_0
	\]
	נשרטט את הפתרון ל־$y$ אשר שקול ל־$x$,
	\begin{otherlanguage}{english}
		\begin{center}

						xmin=-4,

						axis y line=middle,
						axis x line=middle,
						restrict x to domain=-4:4,
						restrict y to domain=-4:4,

						\edef\X{\k/2}

					}
				\end{axis}
			\end{tikzpicture}
		\end{center}
	\end{otherlanguage}
	והפעם נקבל ש־$0$ לא נקודת שיווי משקל באף כיוון.
\end{solution}

\question{}
תהי $F : \Omega \to \RR^n$ ליפשיץ מקומית ונגדיר את בעיית תנאי ההתחלה,
\[
	\begin{cases}
		x'(t) = F(x(t)) \\
		x(0) = x_0
	\end{cases}
	\tag{1}
\]
נניח כי כל הקורדינטות של $F$ אי־שליליות.
תהי $U \subseteq \Omega$ כך שגם $\bar{U} \subseteq \Omega$ ונסמן את הפתרון המקסימלי של $(1)$ ב־$x$.
נראה כי אחת מהאפשרויות הבאה מתקיימת:
\begin{enumerate}
	\item $x$ מגיע לשפה של $U$ בזמן סופי
	\item $\lim_{t \to \infty} x(t) \in \bar{U}$ ונקודה זו היא נקודת שיווי משקל של $F$
\end{enumerate}
\begin{proof}
	מטריכוטומיה או שהמרחק של $x$ מהשפה של $\Omega$ שואף לאפס ולכן בפרט $x$ חוצה את $U$ בזמן סופי כלשהו, או ש־$\lVert x(t) \rVert \to \infty$ ושוב בפרט מחסימות $U$ היא עוברת אותה, או ש־$I$ לא חסום מימין.

	נשאר אם כך להראות שאם $I$ לא חסום מימין אז $\lim_{t \to \infty} x(t) \in \overline{U}$. \\
	נבחין כי $x(t) \in \overline{U}$ לכל $t \ge 0$ מההנחה, אבל $\overline{U}$ חסומה וסגורה ו־$x$ מונוטונית עולה בכל ציר,
	לכן משימוש במשפט התכנסות חסומה ומונוטונית נקבל שכל ציר מתכנס בנפרד ובסך הכול $x(t) \to z$ לאיזושהי $z \in \overline{U}$. \\
	$F$ היא ליפשיץ מקומית ולכן משמרת גבולות, כלומר $F(x(t))$ מתכנסת ולכן $x'(t)$ אף היא מתכנסת כאשר $t \to \infty$, ומטענה נוספת מאינפי 1 נקבל שבמצב זה גם $x'(t) \to 0$ בלבד.
	מצאנו אם כך של־$F$ יש גבול חלקי $0$ ב־$z$, אבל היא רציפה ולכן $\lim_{x \to z} F(x) = 0$ בלבד, כלומר $F(z) = 0$ ו־$z$ היא נקודת שיווי משקל.
\end{proof}

\subquestion{}
נוכיח את שיטת הורדת הסדר:
תהי משוואה $a(t) x''(t) + b(t) x'(t) + c(t) x(t) = 0$, ונניח ש־$y(t)$ פותרת אותה. \\
נראה שפונקציה מהצורה $z(t) = v(t) y(t)$ פותרת את המשוואה אם ורק אם $v'(t)$ פותרת משוואה מסדר ראשון.
\begin{proof}
	על־ידי גזירת $z$ נקבל,
	\[
		z'
		= a z'' + b z' + c z
		= a (v'' y + 2 v' y' + v y'') + b (v' y + v y') + c v y
		= a (v'' y + 2 v' y') + b (v' y)
	\]
	כאשר השתמשנו בעובדה ש־$y$ פתרון, לכן הפונקציה $v'$ מקיימת את,
	\[
		a y x' + 2 a y' x + b y x = 0
	\]
	אם ורק אם $v y$ מקיימת את המשוואה הראשונה ישירות משוויון.
\end{proof}

\subquestion{}
נשתמש בשיטת הורדת הסדר עבור המשוואה $x'' = c^2 x$ על־ידי שימוש בפתרון $y(t) = e^{c t}$.
\begin{solution}
	אנו רוצים למצוא $z(t) = v(t) y(t)$ המקיימת אף היא את המשוואה.
	לצורך כך נרצה לפתור את המשוואה שמצאנו בסעיף הקודם, נציב בה את $y(t) = e^{c t}$ ואת,
	\[
		a(t) = 1,
		\quad
		b(t) = 0,
		\quad
		c(t) = -c^2
	\]
	ונקבל,
	\[
		c^{c t} x'(t) + 2 c e^{c t} x(t) + 0 = 0
	\]
	או לאחר הוצאת גורם משותף והחלפת אגף,
	\[
		x'(t) = -2c\ x(t)
	\]
	והפתרון שלה הוא כמובן $t \mapsto e^{-2c t}$.
	מצאנו אם כך שהפונקציה $v'$ מקיימת $v'(t) = e^{-2c t}$, אם כך,
	\[
		v(t)
		= \int_0^t v'(s)\ ds
		= \frac{1}{-2c} (e^{-2c t} - 1) + C
		= -\frac{1}{2c} e^{-2c} + C
	\]
	היא צורה כללית שפותרת את המשוואה, בפרט,
	\[
		z(t)
		= v(t) y(t)
		= (-\frac{1}{2c} e^{-2c} + C) e^{c t}
		= C e^{c t} - \frac{1}{2 c} e^{-c t}
	\]
	מהווה פתרון נוסף של המשוואה המקורית.
	נבחין כי זוהי משוואה הומוגנית ולכן גם סגורה תחת מכפלה בסקלר, וכן שהיא סגורה לחיבור פתרונות ולכן נסיק שגם $z(t) = e^{-ct}$ פתרון של המשוואה, ואף ש־$\Sp\{ e^{c t}, e^{-c t} \}$ קבוצה של פתרונות של המשוואה.
\end{solution}

\subquestion{}
משוואת אוילר:
נגדיר את משוואת אוילר כמשוואה הלינארית,
\[
	a t^2 x''(t) + b t x'(t) + c x(t) = 0
\]
ונרצה למצוא $r$ עבורו $t^r$ פתרון של המשוואה.

\subsubsection{i}
נמצא פתרון למשוואה,
\[
	t^2 x''(t) + 7 t x'(t) + 5 x(t) = 0
\]
\begin{solution}
	אם $x(t) = t^r$ אז,
	\[
		x'(t)
		= r t^{r - 1},
		\quad
		x''(t)
		= r(r - 1) t^{r - 2}
	\]
	נציב ונקבל,
	\[
		r (r - 1) t^r + 7 r t^r + 5 t^r = 0
	\]
	ונקבל או $t^r = 0$ לכל $t$ או,
	\[
		r^2 + 6r + 5
		= (r + 1) (r + 5)
		= 0
	\]
	נסיק אם כך ש־$t^{-1}, t^{-5}$ הם שני פתרונות של המשוואה הנתונה.
	בהתאם נקבל את קבוצת הפתרונות,
	\[
		\Sp\{ t^{-1}, t^{-5} \}
	\]
	נעיר שזוהי קבוצת כל הפתרונות שכן מימד מרחב הפתרונות הוא $2$ לפי משפט מהתרגול.
\end{solution}

\subsubsection{ii}
נמצא פתרון כללי למשוואה,
\[
	t^2 x''(t) - 3t x'(t) + 4 x(t) = 0
\]
\begin{solution}
	הפעם נקבל באותו תהליך בדיוק,
	\[
		t^r (r^2 - r  - 3r + 4) = 0
	\]
	ושוב $t \not\equiv 0$ ולכן $t^r$ לא מתאפס ונובע,
	\[
		r^2 - 4r + 4
		= (r - 2)(r - 2)
		= 0
	\]
	ולכן $t^2$ הוא פתרון של המשוואה.

	עכשיו נוכל להשתמש בשיטת הורדת הסדר עם $y(t) = t^2$,
	נסמן ונחשב לקראת הצבה,
	\[
		a(t) = t^2,
		\quad
		b(t) = -3t,
		\quad
		c(t) = 4,
		\quad
		y'(t) = 2t
	\]
	ולכן $z(t) = v(t) t^2$ פתרון אם $v'$ פותרת את,
	\[
		t^2 \cdot t^2 x' + 2 t^2 \cdot 2t x + (-3t) \cdot t^2 x
		= t^3 (t x'(t) + x(t))
		= 0
	\]
	שוב נוכל לצמצם ונקבל את המשוואה $t x'(t) + x(t) = 0$.
	נקבל אם כך על־ידי החלפת אגפים,
	\[
		\frac{x'}{x} = -\frac{1}{t}
	\]
	כלומר זוהי משוואה לוגריתמית ולכן $x(t) = \frac{1}{t}$ מהווה פתרון (יחיד) שלה.
	בהתאם $v'(t) = \frac{1}{t}$ ונוריד את המינוס בשל סגירות לכפל בסקלר,
	\[
		v(t)
		= \int \ln |s|\ ds
		= \ln |t|
	\]
	אז גם $z(t) = v(t) y(t) = t^2 \ln |t|$ פתרון של המשוואה וקבוצת הפתרונות הכללית היא,
	\[
		\Sp\{ t^2, \ln |t| \}
	\]
	בדיוק.
\end{solution}

\question{}
נגדיר את המערכת,
\[
	\begin{pmatrix}
		x' \\
		y'
	\end{pmatrix}
	= \begin{pmatrix}
		x + xy - \sqrt{x^2 + y^2} (x + y) \\
		y - x^2 + \sqrt{x^2 + y^2} (x - y)
	\end{pmatrix}
\]
ונבחין כי $(1, 0)$ נקודת שיווי משקל של המערכת.

\subquestion{}
יהי $(x(t), y(t))$ פתרון, ונסמן $(r(t), \varphi(t))$ הפתרון בקורדינטות פולריות שלו, כלומר,
\[
	x(t) = r(t) \cos \varphi(t),
	\qquad
	y(t) = r(t) \sin \varphi(t)
\]
נראה שמתקיים,
\[
	r'(t)
	= r(t) (1 - r(t)),
	\qquad
	\varphi'(t)
	= r(t) (1 - \cos \varphi(t))
\]
\begin{proof}
	מתקיים,
	\[
		x'
		= x + xy - \sqrt{x^2 + y^2} (x + y)
		= r \cos \varphi - r^2 \cos \varphi \sin \varphi - r (r \cos \varphi + r \sin \varphi)
	\]
	וכן,
	\[
		r = \sqrt{x^2 + y^2}
	\]
	ולכן,
	\[
		r'
		= \frac{2x x' + 2y y'}{2 \sqrt{x^2 + y^2}}
		= \frac{1}{r} (x x' + y y')
	\]
	ולכן לפי השוויונות הנתונים,
	\begin{align*}
		r r'
		& = x(x + xy - \sqrt{x^2 + y^2} (x + y)) + y(y - x^2 + \sqrt{x^2 + y^2} (x - y)) \\
		& = x^2 + y^2 - \sqrt{x^2 + y^2} x^2 + \sqrt{x^2 + y^2} - y^2 \\
		& = r^2 - r^3
	\end{align*}
	ונסיק ש־$r' = r (1 - r)$.

	ידוע לנו כי $\tan \varphi(t) = \frac{x(t)}{y(t)}$ ולכן,
	\[
		(\tan \varphi)'
		= \frac{\varphi'}{\cos^2 \varphi}
		= \frac{x' y - x y'}{x^2}
		= \frac{x' y - x y'}{r^2 \cos^2 \varphi}
	\]
	ולכן,
	\begin{align*}
		\varphi' r^2
		& = x' y - x y' \\
		& = (x + xy - \sqrt{x^2 + y^2} (x + y)) y - x (y - x^2 + \sqrt{x^2 + y^2} (x - y)) \\
		& = x y^2 - \sqrt{x^2 + y^2} (xy + y^2) + x^3 - \sqrt{x^2 + y^2} (x^2 - xy) \\
		& = x (y^2 + x^2) - \sqrt{x^2 + y^2} (x^2 + y^2) \\
		& = r^2 \cdot r \cos \varphi - r^3
	\end{align*}
	ולאחר העברת אגפים נקבל $\varphi' = r (1 - \cos \varphi)$.
\end{proof}

\subquestion{}
נראה שקיימת סביבה $(1, 0) \in U$ כך שלכל $(a, b) \in U$ הפתרון למשוואה עם תנאי ההתחלה $(a, b)$ מתכנס ל־$(1, 0)$ ב־$t \to \infty$.
\begin{proof}
	תחילה נבחין כי בקורדינטות פולריות הנקודה $(1, 0)$ מתורגמת ל־$(1, 0)$, ונגדיר את $U = B((1, 0), 1)$ בקורדינטות פולריות, כלומר קבוצה זו מתורגמת לקבוצה פתוחה כלשהי סביב $(1, 0)$ ב־$\RR^2$. \\
	אם $(a, b) \in U$ אז $0 < r(0) < 2$ ולכן אם $0 < r < 1$ אז $r' > 0$ ואם $1 < r < 2$ אז $1 - r < 0$ ולכן גם $r' < 0$, בשני המקרים הפונקציה תהיה מונוטונית מיחידות ומחסימות על־ידי $1$ נקבל $r(t) \to 1$.

	נבחין כי בהנחה שלנו גם $0 < \varphi(0) < 2$ ולכן $1 - \cos \varphi$ מקיים אותה תכונה ומתקיים $\varphi(t) \to 0$ תמיד, ונסיק שאכן $x(t)$ המקיימת $x(0) = (a, b) \in U$ תקיים גם $\lim_{t \to \infty} x(t) = (1, 0)$.
\end{proof}

\subquestion{}
נראה ש־$(1, 0)$ לא נקודת שיווי משקל יציבה.
\begin{proof}
	תהי נקודה $(\cos \alpha, \sin \alpha) \in \RR^2$, ותהי פונקציה $x$ המקיימת את המשוואה עם נקודת התחלה זו, אז מתקיים עבורה $r(0) = 1, \varphi(0) = \alpha$.
	נציב במשוואות הפולריות ונקבל,
	\[
		r'(0)
		= 1 \cdot (1 - 1)
		= 0,
		\quad
		\varphi'(0)
		= 1 - \cos \alpha
	.\]
	מיחידות נסיק שמתקיים $r(t) = 1$ לכל $t \ge 0$.
	בנוסף גם $1 - \cos \alpha > 0$ לכל $\alpha \ne 2 \pi k$, ולכן $\varphi' > 0$ והפונקציה מונוטונית עולה ומתכנסת ל־$2 \pi$ לכל $0 < \alpha < 2 \pi$.
	נסיק אם כך שלכל $\varepsilon > 0$ שנבחר ולכל $\delta > 0$ נוכל לבחור זווית מספיק קטנה וחיובית כך ש־$(a, b) \in B((1, 0), \delta)$ אבל $x$ תנוע על מעגל היחידה נגד כיוון הנקודה $(1, 0)$, ובכך תסתור את הגדרת היציבות.
\end{proof}
